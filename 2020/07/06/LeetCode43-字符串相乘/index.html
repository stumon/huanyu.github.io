LeetCode 43 字符串相乘

题目

给定两个以字符串形式表示的非负整数 num1 和 num2，返回 num1 和 num2 的乘积，它们的乘积也表示为字符串形式

>示例 2:
>输入: num1 = "123", num2 = "456"
>输出: "56088"

思路

根据我们日常乘法运算的规则，列竖式，一个个乘，然后把这些结果相加，就得到了最终结果

代码

class Solution {
    public String multiply(String num1, String num2) {
        int len1=num1.length();
        int len2=num2.length();
        StringBuilder res=new StringBuilder();
        if(len1==0 || len2==0) return res.toString();
        if(num1.equals("0")||num2.equals("0")) return "0";//如果两个字符串有一个为零，那么就返回0
        for(int i=0;i<len1;i++)
        {
            StringBuilder tmp=new StringBuilder();//记录每一次运算后的结果
            int m=num1.charAt(len1-1-i)-'0';//从num1中取一个数字，取乘num2的每一数字
            int carry=0;//进位
            int count=0;
            for(int j=0;j<len2;j++)
            {
                int n=num2.charAt(len2-1-j)-'0';//从个位
                count=n*m+carry;//num1中某一位和num2中某一位计算出的结果，再加上进位
                carry=count/10;//进位
                tmp.append(count%10);//把每一次运算的结果保存到tmp中
            }
            if(carry>0) tmp.append(carry);//如果进位不为0，就添加进去
            tmp.reverse();//tmp中保存的结果是逆序的，所以需要翻转一下
            for(int k=0;k<i;k++) tmp.append(0);//添加0，个位乘一个数字，和十位乘数字，十位的结果要加0，不明白的话列一个竖式乘法就懂了
            res=new StringBuilder(addStr(res.toString(),tmp.toString()));   //把这一次的结果和之前结果进行相加
        }
        return res.toString();
    }
    //把两个字符串进行相加
    String addStr(String num1, String num2)
    {
         int len1=num1.length(),len2=num2.length();
        if(len1==0) return num2;
        if(len2==0) return num1;
        int len3=Math.max(len1,len2);
        StringBuilder sb=new StringBuilder();
        int carry=0;
        for(int i=0;i<len3;i++)
        {
            int n1=i>len1-1?0:num1.charAt(len1-1-i)-'0';
            int n2=i>len2-1?0:num2.charAt(len2-1-i)-'0';
            int n3=n1+n2+carry;
            carry=n3/10;
            sb.append(n3%10);
        }
        if(carry>=1) sb.append("1");
        return sb.reverse().toString();
    }
}
