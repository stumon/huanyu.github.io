LeetCode 227. 基本计算器 II

题目

字符串表达式仅包含非负整数，+， - ，*，/ 四种运算符和空格 。 整数除法仅保留整数部分。

示例:

>输入: " 3+5 / 2 "
>输出: 5

思路

中缀转后缀，没有括号，但是有乘除，所以要比较运算符的优先级

使用栈+链表：栈用来维护运算符，链表用来记录后缀表达式

代码

class Solution {
    public int calculate(String s) {
        return calBackExp(transferToBackExp(s));
    }
    //计算后缀表达式
    private int calBackExp(LinkedList<String> list)
    {
        Stack<Integer> stack=new Stack<>();
        for(String s:list)
        {
            if(isNumerical(s))
            {
                stack.push(Integer.parseInt(s));
            }else if(s.equals("+") || s.equals("-") || s.equals("*") || s.equals("/")){
                int n1=stack.pop();
                int n2=stack.pop();
                int n3=0;
                if(s.equals("+")){
                    n3=n1+n2;
                }else if(s.equals("-")){
                    n3=n2-n1;
                }else if(s.equals("*")){
                    n3=n1*n2;
                }else{
                    n3=n2/n1;
                }
                stack.push(n3);
            }
        }
    return stack.pop();
    }

	//中缀转后缀
    private LinkedList<String> transferToBackExp(String exp)
    {
        Stack<String> stack=new Stack<>();
        LinkedList<String> list=new LinkedList<>();
        int len=exp.length();
        for(int i=0;i<len;i++)
        {
            char c=exp.charAt(i);
            if(isDigit(c))
            {
                int k=i+1;
                for(;k<len && isDigit(exp.charAt(k));k++){}//出现连续数字时，比如6555+4564，直接计算数字的起始位置和终止位置
                list.add(exp.substring(i,k));
                i=k-1;
            }else if(c=='*' || c=='/')
            {
                //如果当前运算符是乘除，那么栈顶运算符只要不是加减，就必须出栈，直到是加减或者为空为止
                while(!stack.empty() && (stack.peek().equals("*") || stack.peek().equals("/")))//比较优先级
                {
                    list.add(stack.pop());
                }
                stack.push(c+"");
            }else if(c=='+' || c=='-'){//如果当前运算符是加减，因为加减运算符是最低的，没有比它低的，所以就必须出栈，直到栈为空。
                while(!stack.empty()){
                    list.add(stack.pop());
                }
                stack.push(c+"");
            }
        }
        while(stack.size()>0) list.add(stack.pop());//如果栈里面还有元素，就依次添加到链表中
        return list;
    }
    boolean isDigit(char c)
    {
        if(c-'0'>=0 && c-'0'<=9) return true;
        return false;
    }
    boolean isNumerical(String s)
    {
        for(int i=0;i<s.length();i++)
        {
            char c=s.charAt(i);
            if(isDigit(c)==false) return false;
        }
        return true;
    }

}
